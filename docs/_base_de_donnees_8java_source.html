53 
54  List<String> propositions = Arrays.asList(prop1, prop2, prop3, prop4);
55 
56  int nombreCarateres = (question + prop1 + prop2 + prop3 + prop4).length();
57  int tempsReponse = Parametres.getParametres().estCalulAutomatiqueDuTempsDeReponse() ? getTempsReponse(-1, nombreCarateres) : Parametres.getParametres().getTempsReponse();
58 
59  return new Question(idQuestion, question, propositions, tempsReponse);
