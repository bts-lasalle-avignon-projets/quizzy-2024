231 void CommunicationBluetooth::decoderTrame(QString trame)
232 {
233  trame.remove(0, 1); // supprime le $
234  trame.remove(trame.size() - 1, 1); // supprime le \n
235  QStringList champs = trame.split(SEPARATEUR_DE_CHAMPS);
236  qDebug() << Q_FUNC_INFO << "champs" << champs;
237  QChar typeTrame = champs[TYPE_TRAME].at(0);
238 
239  switch(typeTrame.toLatin1())
240  {
241  case 'L': // Lancer un quiz : $L\n
242  emit debutQuiz();
243  break;
244  case 'I': // Un participant au quiz : $I;PID;NOM DU JOUEUR\n
245  emit nouveauParticipant(champs[PID_JOUEUR], champs[NOM_JOUEUR]);
246  break;
247  case 'G': // Une question :
248  // $G;LIBELLE;R1;R2;R3;R4;NUMERO_REP_VALIDE;TEMPS\n
249  {
250  QStringList propositions;
251  propositions << champs[PROPOSITION_A] << champs[PROPOSITION_B]
252  << champs[PROPOSITION_C] << champs[PROPOSITION_D];
253  emit nouvelleQuestion(champs[LIBELLE],
254  propositions,
255  champs[NUMERO_REPONSE_VALIDE].toInt(),
256  champs[TEMPS].toInt());
257  break;
258  }
259  case 'T': // Signaler le démarrage (top) d’une question : $T\n
260  emit debutQuestion();
261  break;
262  case 'U': // Réponse choisie par un joueur :
263  // $U;PID_JOUEUR;NUMÉRO_REPONSE;TEMPS_REPONSE\n
264  emit choixReponse(champs[PID_JOUEUR],
265  champs[NUMERO_REPONSE_PARTICIPANT].toInt(),
266  champs[TEMPS_REPONSE_PARTICIPANT].toInt());
267  break;
268  case 'H': // Afficher la réponse : $H\n
269  emit bonneReponse();
270  break;
271  case 'S': // Passer à la question suivante : $S\n
272  emit questionSuivante();
273  break;
274  case 'P': // Revenir à la question précédente : $P\n
275  emit questionPrecedente();
276  break;
277  case 'F': // Finir un quiz : $F\n
278  emit finQuiz();
279  break;
280  default:
281  qDebug() << Q_FUNC_INFO << "type de trame inconnu !";
282  break;
283  }
